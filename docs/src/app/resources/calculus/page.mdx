## Derivatives

$$
\begin{align}
  \dv{}{x}c = 0
\end{align}
$$
$$
\begin{align}
  \dv{}{x}\left[f(x)+g(x)\right] = f'(x) + g'(x)
\end{align}
$$
$$
\begin{align}
  \dv{}{x}\left[f(x)g(x)\right] = f'(x)g(x) + f(x)g'(x)
\end{align}
$$
$$
\begin{align}
  \dv{}{x}\frac{f(x)}{g(x)} = \frac{g(x)f'(x)-f(x)g'(x)}{[g(x)]^2}
\end{align}
$$
$$
\begin{align}
  \dv{}{x}f(g(x)) = f'(g(x))g'(x)
\end{align}
$$
$$
\begin{align}
  \dv{}{x}x^n = nx^{n-1}
\end{align}
$$

### Exponential and Logarithmic Functions

$$
\begin{align}
  \dv{}{x} e^x = e^x
\end{align}
$$
$$
\begin{align}
  \dv{}{x} a^x = a^x \ln a
\end{align}
$$
$$
\begin{align}
  \dv{}{x} \ln |x| = \frac{1}{x}
\end{align}
$$
$$
\begin{align}
  \dv{}{x} \log_a x = \frac{1}{x \ln a}
\end{align}
$$

### Trigonometric Functions

$$
\begin{align}
  \dv{}{x} \sin x = \cos x
\end{align}
$$
$$
\begin{align}
  \dv{}{x} \cos x = -\sin x
\end{align}
$$
$$
\begin{align}
  \dv{}{x} \tan x = \sec^2 x
\end{align}
$$
$$
\begin{align}
  \dv{}{x} \csc x = -\csc x \cot x
\end{align}
$$
$$
\begin{align}
  \dv{}{x} \sec x = \sec x \tan x
\end{align}
$$
$$
\begin{align}
  \dv{}{x} \cot x = -\csc^2 x
\end{align}
$$
